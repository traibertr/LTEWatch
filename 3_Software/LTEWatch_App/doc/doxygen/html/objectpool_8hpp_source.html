102 explicit ObjectPool(size_t numberOfObjectsToPreallocate, uint8_t * * memory = NULL)
103 {
104 // If the objects should be placed into a given memory region.
105 if (memory)
106 {
107 // Create the object pool.
108 _objectPool = new (static_cast<void *>(*memory)) T[numberOfObjectsToPreallocate];
109 *memory += numberOfObjectsToPreallocate * sizeof(T);
110
111 // Create the reference count variables.
112 _referenceCounts = new (static_cast<void *>(*memory)) int8_t[numberOfObjectsToPreallocate];
113 *memory += numberOfObjectsToPreallocate;
114
115 // Remember that we should not delete the memory afterwards.
116 _dynamicMemoryUsed = false;
117 printk("-ObjectPool -> INFO : %2d objects of %2d bytes preallocated in memory...\r\n", numberOfObjectsToPreallocate, sizeof(T));
118 }
119
120 // Use systems malloc()/new() functions to allocate memory.
121 else
122 {
123 // Create the object pool.
124 _objectPool = new T[numberOfObjectsToPreallocate];
125
126 // Create the reference count variables.
127 _referenceCounts = new int8_t[numberOfObjectsToPreallocate];
128
129 // Remember that we should delete the memory afterwards.s
130 _dynamicMemoryUsed = true;
131 printk("-ObjectPool -> INFO : %2d objects of %2d bytes preallocated in dynamic memory...\r\n", numberOfObjectsToPreallocate, sizeof(T));
132 }
133
134 // Initialize the reference count and total free objects variables on success.
135 __ASSERT(_objectPool && _referenceCounts,"Error");
136 if (_objectPool && _referenceCounts)
137 {
138 memset(_referenceCounts, 0, numberOfObjectsToPreallocate);
139 _objectCount = numberOfObjectsToPreallocate;
140 _availableObjectCount = numberOfObjectsToPreallocate;
141 printk("-ObjectPool -> INFO : %d of %d objects available (size: %d)\r\n", _availableObjectCount, _objectCount, sizeof(T));
142 }
143
144 // Something went wrong, so we can not offer any object.
145 else
146 {
147 if (_objectPool)
148 delete[] _objectPool;
149 if (_referenceCounts)
150 delete[] _referenceCounts;
151 _dynamicMemoryUsed = false;
152 _objectCount = 0;
153 _availableObjectCount = 0;
154 }
